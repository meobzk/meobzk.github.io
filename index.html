    <!-- Nhúng Firebase SDK sử dụng ES Modules từ CDN -->

    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js";

        import { getDatabase, ref, set, onValue } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-database.js";


        // Cấu hình Firebase của dự án meobzk

        const firebaseConfig = {

            apiKey: "AIzaSyAXwW5V-Yfs4wVFZpM7Z2uKJQjl_BC5tHw",

            authDomain: "meobzk.firebaseapp.com",

            databaseURL: "https://meobzk-default-rtdb.asia-southeast1.firebasedatabase.app",

            projectId: "meobzk",

            storageBucket: "meobzk.appspot.com",

            messagingSenderId: "226897249765",

            appId: "1:226897249765:web:1800a362b02e6d690a1b0c"

        };


        // Khởi tạo Firebase ứng dụng Web

        const app = initializeApp(firebaseConfig);

        const db = getDatabase(app);


        // Tạo tham chiếu đến một thư mục tên là 'message' trên Database

        const dbRef = ref(db, 'message');


        // 1. Lắng nghe thay đổi dữ liệu thời gian thực (Read)

        onValue(dbRef, (snapshot) => {

            const data = snapshot.val();

            document.getElementById('data-display').innerText = data ? data : "Chưa có dữ liệu";

        });


        // 2. Gửi dữ liệu mới lên khi người dùng nhấn nút (Write)

        document.getElementById('btn-send').addEventListener('click', () => {

            const inputValue = document.getElementById('input-data').value;

            if (inputValue.trim() !== "") {

                set(dbRef, inputValue)

                    .then(() => {

                        alert("Cập nhật dữ liệu thành công!");

                        document.getElementById('input-data').value = "";

                    })

                    .catch((error) => {

                        console.error("Lỗi khi ghi dữ liệu: ", error);

                        alert("Gặp lỗi, vui lòng kiểm tra Security Rules!");

                    });

            }

        });
